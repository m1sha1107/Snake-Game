#capture video opecv
vid = cv.VideoCapture(0)
s=0.5
while True:
    try:
        bool, frame = vid.read()
        res = cv.resize(frame, (int(frame.shape[1] * s), int(frame.shape[0] * s)), interpolation=cv.INTER_AREA)
        cv.imshow('yes', res)
        cv.waitKey(20)
    except:
        break
vid.release()
cv.destroyAllWindows()
